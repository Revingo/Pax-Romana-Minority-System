culture_map_start_scripted_gui = {
	scope = culture

	effect = {
		set_global_variable = {
			name = culture_map_culture
			value = root
		}

		every_county = {
			if = {
				limit = {
					culture_is_already_in_list = {
						CULTURE = global_var:culture_map_culture
					}
					culture_is_in_range = {
						CULTURE = global_var:culture_map_culture
						LOW_AMOUNT = 90
						HIGH_AMOUNT = 101
					}
				}

				set_color_from_title = title:d_PAX_color_100
			}
			else_if = {
				limit = {
					culture_is_already_in_list = {
						CULTURE = global_var:culture_map_culture
					}
					culture_is_in_range = {
						CULTURE = global_var:culture_map_culture
						LOW_AMOUNT = 80
						HIGH_AMOUNT = 90
					}
				}

				set_color_from_title = title:d_PAX_color_90
			}
			else_if = {
				limit = {
					culture_is_already_in_list = {
						CULTURE = global_var:culture_map_culture
					}
					culture_is_in_range = {
						CULTURE = global_var:culture_map_culture
						LOW_AMOUNT = 70
						HIGH_AMOUNT = 80
					}
				}

				set_color_from_title = title:d_PAX_color_80
			}
			else_if = {
				limit = {
					culture_is_already_in_list = {
						CULTURE = global_var:culture_map_culture
					}
					culture_is_in_range = {
						CULTURE = global_var:culture_map_culture
						LOW_AMOUNT = 60
						HIGH_AMOUNT = 70
					}
				}

				set_color_from_title = title:d_PAX_color_70
			}
			else_if = {
				limit = {
					culture_is_already_in_list = {
						CULTURE = global_var:culture_map_culture
					}
					culture_is_in_range = {
						CULTURE = global_var:culture_map_culture
						LOW_AMOUNT = 50
						HIGH_AMOUNT = 60
					}
				}

				set_color_from_title = title:d_PAX_color_60
			}
			else_if = {
				limit = {
					culture_is_already_in_list = {
						CULTURE = global_var:culture_map_culture
					}
					culture_is_in_range = {
						CULTURE = global_var:culture_map_culture
						LOW_AMOUNT = 40
						HIGH_AMOUNT = 50
					}
				}

				set_color_from_title = title:d_PAX_color_50
			}
			else_if = {
				limit = {
					culture_is_already_in_list = {
						CULTURE = global_var:culture_map_culture
					}
					culture_is_in_range = {
						CULTURE = global_var:culture_map_culture
						LOW_AMOUNT = 30
						HIGH_AMOUNT = 40
					}
				}

				set_color_from_title = title:d_PAX_color_40
			}
			else_if = {
				limit = {
					culture_is_already_in_list = {
						CULTURE = global_var:culture_map_culture
					}
					culture_is_in_range = {
						CULTURE = global_var:culture_map_culture

					}

			}

					culture_is_already_in_list = {
						CULTURE = global_var:culture_map_culture
					}
					culture_is_in_range = {
						CULTURE = global_var:culture_map_culture
						LOW_AMOUNT = 10
						HIGH_AMOUNT = 20
					}
				}

				set_color_from_title = title:d_PAX_color_20
			}
			else_if = {
				limit = {
					culture_is_already_in_list = {
						CULTURE = global_var:culture_map_culture
					}
					culture_is_in_range = {
						CULTURE = global_var:culture_map_culture
						LOW_AMOUNT = 1
						HIGH_AMOUNT = 10
					}
				}

				set_color_from_title = title:d_PAX_color_10
			}
			else = {
				set_color_from_title = title:d_PAX_color_0
			}
		}
	}

	is_shown = {
		NOT = { has_global_variable = culture_map_culture }
	}
}
