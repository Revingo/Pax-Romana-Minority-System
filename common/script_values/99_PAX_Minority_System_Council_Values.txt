current_culture_value = {
	value = 0
	if = {
		limit = { exists = scope:county }
		scope:county = {
			every_in_list = {
				variable = culture_minorities
				limit = { this.var:culture = scope:councillor_liege.culture }

				add = this.var:amount
			}
		}
	}
}
